START

//INPUT 
INPUT NUM1, NUM2;


//CALCULATING GCD
IF ( NUM1 > NUM2 ){                                // CHECK WHICH NUMBER IS BIGGER TO DIVIDE CORRECTLY
    IF ( NUM1 % NUM2 == 0){
        RETURN GCD IS NUM2;                       // SINCE REMAINDER OF TWO NUMBER IS 0 SO THE SMALLER NUMBER IS THE ONLY GCD.
    }
    ELSE {
        ANSWER = NUM1 % NUM2;                      // CHECK IF REMAINDER IS NON-ZERO SO THE REMAINDER WILL BE THE GCD.
        RETURN ANSWER;
    } 
}
ELSE IF ( NUM2 > NUM1 ){
    IF ( NUM2 % NUM1 == 0){
        RETURN GCD IS NUM1;                       // SINCE REMAINDER OF TWO NUMBER IS 0 SO THE SMALLER NUMBER IS THE ONLY GCD.
    }
    ELSE {
        ANSWER = NUM2 % NUM1;                      // CHECK IF REMAINDER IS NON-ZERO SO THE REMAINDER WILL BE THE GCD.
        RETURN ANSWER;
    } 
}
ELSE {
    RETURN BOTH NUMBERS ARE EQUAL, SO THEIR GCD IS NUM1 AND NUM2 BOTH;
};

//ENDING 
END
